81static void event_handler(void* arg, esp_event_base_t event_base,
82 int32_t event_id, void* event_data)
83{
84 if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_START) {
85 esp_wifi_connect();
86 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_DISCONNECTED) {
87 if (s_retry_num < ESP_MAXIMUM_RETRY) {
88 esp_wifi_connect();
89 s_retry_num++;
90 ESP_LOGI(tm_tag, "retry to connect to the AP");
91 } else {
92 xEventGroupSetBits(s_wifi_event_group, WIFI_FAIL_BIT);
93 }
94 ESP_LOGI(tm_tag,"connect to the AP fail");
95 } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
96 ip_event_got_ip_t* event = (ip_event_got_ip_t*) event_data;
97 ESP_LOGI(tm_tag, "got ip:" IPSTR, IP2STR(&event->ip_info.ip));
98 s_retry_num = 0;
99 xEventGroupSetBits(s_wifi_event_group, WIFI_CONNECTED_BIT);
100 }

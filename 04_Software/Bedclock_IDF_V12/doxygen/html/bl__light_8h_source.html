53/*
54LED algorithm
55
56example:
57- LEDs are currently off
58- maximum brightness is 40
59- target intensity is 2 (=0.16)
60- target color is 2 (orange: r=1.333 g=1.000 b=0.667)
61- light switches on in 20 steps
62
63 red channel is stepping from 0 to 0.16*1.333*40 = 8.433 with steps of 0.422
64green channel is stepping from 0 to 0.16*1.000*40 = 6.325 with steps of 0.316
65 blue channel is stepping from 0 to 0.16*0.667*40 = 4.216 with steps of 0.211
66
67Values of the red channel: 0.00, 0.42, 0.84, 1.26, 1.69, 2.11, 2.53, 2.95, 3.37, etc.
68
69To produce non-integer intensities, a fraction of the 16 LEDs switch to the next value
70*/

73// This function determines the intensity of the LEDs
74uint8_t led_value(double value, uint16_t mask_bit) {
75
76 // Ensure the light goes off when supposed to
77 // to prevent one led remains on
78 if(value<0.07) return 0; // little more than 1/16
79
80 // The order in which the LEDs in the 4x4 grid are switched on
81 const uint16_t SUB_BRIGHTNESS[] = {0x0000, 0x0008, 0x0208, 0x020A, 0x0A0A, 0x2A0A, 0x2A8A, 0x2AAA, 0xAAAA,
82 0xAABA, 0xABBA, 0xABBE, 0xEBBE, 0xEBBF, 0xFBBF, 0xFBFF };
83
84 // Round value used as the intensity of the LED
85 uint8_t i= (int) value;
86
87 // The fraction determines if the LEDs intensity is increased
88 // to approximate the requested average intensity of the panel
89 uint8_t remainder = (uint8_t) 16*(value-i);
90 if( SUB_BRIGHTNESS[remainder] & mask_bit ) i++;
91
92 return i;
93};

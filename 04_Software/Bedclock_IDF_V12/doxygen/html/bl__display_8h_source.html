193void task_display(void *z)
194{
195 bool refresh_display = true;
196 bool animation_still_running = false;
197 common_command_t cmd = CMD_NONE;
198 hp_timer_t sleep_timer; hp_timer_init_ms(&sleep_timer, 20000, false);
199 hp_stepping_float_t screen_y; hp_stepping_float_init(&screen_y, 26, 0, 25, ST_CLICK);
200 hp_stepping_float_t indicator_y; hp_stepping_float_init(&indicator_y, 36, 36, 16, ST_SPRING);
201
202 // display_refresh( (int16_t) screen_y.value, (int16_t) indicator_y.value);
203
204 while(true) {
205
206 if (xQueueReceive(display_queue, &cmd, pdMS_TO_TICKS(100)) == true) {
207 switch(cmd) {
208 case CMD_BTN_LEFT_PRESSED:
209 ESP_LOGI(dp_tag, "item received BTN_LEFT");
210 hp_timer_reset(&sleep_timer);
211 common_increment_selected_item();
212
213 // Move the display to a new target
214 uint8_t selected_item = common_get_selected_item();
215
216 hp_stepping_float_target(&screen_y, Y_POS[selected_item], selected_item==0 ? 20 : 10);
217 if (selected_item>1) hp_stepping_float_target(&indicator_y, Y_IND[selected_item], 6);
218
219 refresh_display = true;
220 break;
221
222 case CMD_BTN_RIGHT_PRESSED:
223 ESP_LOGI(dp_tag, "item received BTN_RIGHT");
224 hp_timer_reset(&sleep_timer);
225
226 switch(common_get_selected_item()) {
227
228 case 0: // Wake up clock
229 ESP_LOGI(dp_tag, "Wake up clock on right button");
230 common_change_selected_item(1);
231 hp_stepping_float_target(&screen_y, Y_POS[1], 8);
232 refresh_display = true;
233 break;
234
235 case 2: // Modify LED intensity
236 ESP_LOGI(dp_tag, "Increase LED intensity");
237 common_increment_led_intensity();
238 queue_send_message(light_queue, CMD_ADJUST_LIGHT);
239 refresh_display = true;
240 break;
241
242 case 3: // Modify LED color
243 ESP_LOGI(dp_tag, "Increase LED color");
244 common_increment_led_color();
245 queue_send_message(light_queue, CMD_ADJUST_LIGHT);
246 refresh_display = true;
247 break;
248
249 case 4: // Modify LED timer
250 ESP_LOGI(dp_tag, "Increase LED timer");
251 common_increment_led_timer();
252 queue_send_message(light_queue, CMD_TIMER_CHANGED);
253 refresh_display = true;
254 break;
255
256 case 5: // Modify display intensity
257 ESP_LOGI(dp_tag, "Modify display intensity");
258 common_increment_led_intensity();
259 refresh_display = true;
260 break;
261
262
263 default:
264 ESP_LOGI(dp_tag, "Item %d cannot be changed ", common_settings.selected_item);
265 } // case CMD_BTN_RIGHT_PRESSED -> switch(common_settings.selected_item) {
266 break;
267
268 case CMD_LIGHT_SWITCHES_ON:
269 ESP_LOGI(dp_tag, "Light switches on, show clock");
270 if( common_get_selected_item()==0 ) {
271 common_change_selected_item(1);
272 hp_stepping_float_target(&screen_y, Y_POS[1], 30);
273 refresh_display = true;
274 }
275
276 if( common_get_selected_item()==1 ) {
277 // Kill the sleep timer, so clock only removed when LED switches off
278 hp_timer_set_lapsed(&sleep_timer);
279 ESP_LOGI(dp_tag, "hp_timer_lapsed = %s ", hp_timer_lapsed(&sleep_timer) ? "yes" : "no");
280 }
281 break;
282
283
284 case CMD_LIGHT_SWITCHES_OFF:
285 ESP_LOGI(dp_tag, "Light switches off, remove clock too");
286 if( common_get_selected_item()==1 ) {
287 common_change_selected_item(0);
288 hp_stepping_float_target(&screen_y, Y_POS[0], 15);
289 refresh_display = true;
290 }
291 break;
292
293 case CMD_RESET_DISPLAY:
294 ESP_LOGI(dp_tag, "Reset display");
295 display_reset();
296 break;
297
298 default:
299 ESP_LOGI(dp_tag, "Received handled button");
300 } // switch(cmd) {
301 } // if (xQueueReceive(display_queue
302
303 if(hp_timer_lapsed(&sleep_timer)) {
304 ESP_LOGI(dp_tag, "Display back to sleep");
305 common_change_selected_item(0);
306 hp_stepping_float_target(&screen_y, Y_POS[0], 15);
307
308 // Check if we need to save settings
309 common_save();
310
311 refresh_display = true;
312 } // if(hp_timer_lapsed(&sleep_timer))
313
314 // Refresh display if minute has changed
315 if(common_settings.selected_item>0) {
316 time_t now = 0;
317 struct tm timeinfo = { 0 };
318 time(&now);
319 localtime_r(&now, &timeinfo);
320 static int prev_min = -1;
321 if( prev_min != timeinfo.tm_min ) {
322 ESP_LOGI(dp_tag, "Update clock due to minute change");
323 prev_min = timeinfo.tm_min;
324 refresh_display=true;
325 }
326 }
327
328 // Also refresh display if animation is still going on
329 // Not a pretty solution, but before introducing animation_still_running,
330 // the last frame of the animation was not shown
331 hp_stepping_float_step(&screen_y);
332 hp_stepping_float_step(&indicator_y);
333 if(!hp_stepping_float_finished(&screen_y)) animation_still_running=true;
334 if(!hp_stepping_float_finished(&indicator_y)) animation_still_running=true;
335 if(animation_still_running) refresh_display = true;
336
337 if(refresh_display) {
338 display_refresh( (int16_t) screen_y.value, (int16_t) indicator_y.value );
339 refresh_display = false;
340 } // if(refresh_display) {
341
342 // After refreshing display, decide if animation has stopped
343 if(hp_stepping_float_finished(&screen_y)) animation_still_running=false;
344 if(hp_stepping_float_finished(&indicator_y)) animation_still_running=false;
345
346 vTaskDelay(pdMS_TO_TICKS(2));
347 } // while true
348} // void task_display(void *z)

92float hp_stepping_float_interpolation(float x, hp_stepping_interpolation_t interpolation) {
93
94 // Translates (0..1) input to (0..1) output according to selected interpolation method
95 float x_;
96
97 switch(interpolation) {
98 case ST_LINEAR:
99 return x;
100
101 case ST_CYCLOID:
102 return x - 1/(2*M_PI)*sin(2*M_PI*x);
103
104 case ST_SPRING:
105 if(x<0.65) {
106 float x_=x/0.65;
107 return 3.6*pow(x_,2)*(1-x_) + 1.2*pow(x_,3);
108 }
109
110 x_ = (x-0.65)/0.35;
111 return 1.2*pow(1-x_,3) + 3.6*x_*pow(1-x_,2) + 3*pow(x_,2)*(1-x_) + pow(x_,3);
112
113 case ST_CLICK:
114 return x-0.4052443*sin(2*M_PI*x)+0.1453085*sin(4*M_PI*x);
115
116 case ST_ACCEL:
117 return pow(x,2);
118
119 case ST_DAMPED:
120 return 1-pow(1-x,2);
121
122 case ST_BOUNCE:
123 if(x<0.5) return pow(2*x,3);
124
125 if(x<0.8) {
126 x_ = (x-0.5)/0.3;
127 return pow(1-x_,3)+3*x_*pow(1-x_,2)*0.75+3*pow(x_,2)*(1-x_)*0.75+pow(x_,3);
128 }
129
130 x_ = (x-0.8)/0.2;
131 return pow(1-x_,3)+3*x_*pow(1-x_,2)*0.9+3*pow(x_,2)*(1-x_)*0.9+pow(x_,3);
132
133 case ST_BOW:
134 return 3*x*pow(1-x,2)*-0.4+pow(x,3);
135
136 case ST_STEP:
137 if(x<0.4) {
138 x_ = x/0.4;
139 return 3.3*pow(x_,2)*(1-x_) + 1.1*pow(x_,3);
140 }
141
142 if(x<0.6) {
143 x_ = (x-0.4)/0.2;
144 return 1.1* pow(1-x_,3) + 3.3*x_*pow(1-x_,2) + 2.85*pow(x_,2)*(1-x_) + 0.95*pow(x_,3);
145 }
146
147 if(x<0.8) {
148 x_ = (x-0.6)/0.2;
149 return 0.95*pow(1-x_,3) + 2.85*x_*pow(1-x_,2) + 3.15*pow(x_,2)*(1-x_) + 1.05*pow(x_,3);
150 }
151
152 x_ = (x-0.8)/0.2;
153 return 1.05*pow(1-x_,3) + 3.15*x_*pow(1-x_,2) + 3*pow(x_,2)*(1-x_) + pow(x_,3);
154
155 default:
156 // Also ST_LINEAR
157 return x;
158 }

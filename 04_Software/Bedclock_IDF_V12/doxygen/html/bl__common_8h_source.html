170void common_init() {

172 display_queue = xQueueCreate(20, sizeof(common_command_t));
173 light_queue = xQueueCreate(20, sizeof(common_command_t));
174
175 ESP_LOGI(cm_tag, "Reading variables from non volatile storage");
176 esp_err_t err = nvs_flash_init();
177 if (err == ESP_ERR_NVS_NO_FREE_PAGES || err == ESP_ERR_NVS_NEW_VERSION_FOUND) {
178 // NVS partition was truncated and needs to be erased
179 // Retry nvs_flash_init
180 ESP_ERROR_CHECK(nvs_flash_erase());
181 err = nvs_flash_init();
182 }
183 ESP_ERROR_CHECK( err );
184
185 nvs_handle_t nvs_handle;
186 ESP_ERROR_CHECK(nvs_open(nvs_namespace, NVS_READWRITE, &nvs_handle));
187
188 if (xSemaphoreTake(mutex_change_settings, portMAX_DELAY) == pdTRUE) {
189
190 // Read
191 err = nvs_get_u8(nvs_handle, "led_intensity", &common_settings.led_intensity);
192
193 switch (err) {
194 case ESP_OK:

250void common_save() {
251 // Only save settings if settings were changed
252 // single boolean can be read without using semaphore
253 if(!common_settings.settings_changed) return;
254
255 ESP_LOGI(cm_tag, "Saving variables to non volatile storage");
256
257 esp_err_t err = nvs_flash_init();
258 if (err == ESP_ERR_NVS_NO_FREE_PAGES || err == ESP_ERR_NVS_NEW_VERSION_FOUND) {
259 // NVS partition was truncated and needs to be erased
260 // Retry nvs_flash_init
261 ESP_ERROR_CHECK(nvs_flash_erase());
262 err = nvs_flash_init();

264 ESP_ERROR_CHECK( err );
265
266 nvs_handle_t nvs_handle;
267 nvs_open(nvs_namespace, NVS_READWRITE, &nvs_handle);
268
269 if (xSemaphoreTake(mutex_change_settings, portMAX_DELAY) == pdTRUE) {
270 // Write
271 nvs_set_u8(nvs_handle, "led_intensity", common_settings.led_intensity );
272 nvs_set_u8(nvs_handle, "led_color", common_settings.led_color );

274 err = nvs_set_u8(nvs_handle, "disp_int", common_settings.display_intensity);
